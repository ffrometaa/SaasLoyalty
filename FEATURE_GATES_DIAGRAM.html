Code Usage Examples
Example 1 — FeatureGate component (wraps any UI)
// Renders children if plan has feature, otherwise shows lock UI
<FeatureGate
  plan={tenant.plan}
  feature="analytics_heatmap"
>
  <HeatmapChart data={data} />
</FeatureGate>

// silent=true renders nothing instead of lock UI (good for buttons)
<FeatureGate
  plan={tenant.plan}
  feature="analytics_export"
  silent
>
  <ExportButton />
</FeatureGate>
Example 2 — useFeatureGate hook (imperative checks)
const { hasFeature, memberLimitReached } = useFeatureGate({
  plan: tenant.plan,
  currentMemberCount: members.length,
});

// Conditional render without wrapping component
if (!hasFeature('booking_integrations')) return null;

// Disable a button when at member limit
const atLimit = memberLimitReached();
<button disabled={atLimit}>Add Member</button>
Example 3 — Server-side guards (API routes / Server Actions)
// In a Route Handler — enforce feature access
await requireFeature(tenantId, 'analytics_export');

// In POST /api/members — enforce member slot
try {
  await requireMemberSlot(tenantId);
} catch (err) {
  return NextResponse.json({ error: err.message }, { status: 403 });
}

// In POST /api/campaigns — enforce campaign slot
await requireCampaignSlot(tenantId);
Example 4 — LimitWarningBanner (usage warnings)
// Shows amber warning at 80% usage, red banner at 100%
<LimitWarningBanner
  plan={tenant.plan}
  type="members"
  current={memberCount}
  warningThreshold={80}
/>

// For campaigns
<LimitWarningBanner
  plan={tenant.plan}
  type="campaigns"
  current={campaignsThisMonth}
/>

// Returns null for Scale plan (no limits) — safe to always render
